    <script>

        const samples = document.querySelectorAll('.sample');

        const submitBtn = document.querySelector('.submit-btn');

        const results = document.querySelector('.results');

        const correctAnswers = ['ai', 'human', 'ai', 'human', 'human'];

        let userAnswers = [];

 

        // Add click handlers to choice buttons

        samples.forEach((sample, index) => {

            const buttons = sample.querySelectorAll('.choice-btn');

            buttons.forEach(btn => {

                btn.addEventListener('click', () => {

                    // Remove selected class from all buttons in this sample

                    buttons.forEach(b => b.classList.remove('selected'));

                    // Add selected class to clicked button

                    btn.classList.add('selected');

                    // Store the answer

                    userAnswers[index] = btn.dataset.choice;

                   

                    // Check if all questions are answered

                    if (userAnswers.filter(answer => answer).length === 5) {

                        submitBtn.disabled = false;

                    }

                });

            });

        });

 

        function checkAnswers() {

            let score = 0;

           

            samples.forEach((sample, index) => {

                const buttons = sample.querySelectorAll('.choice-btn');

                const explanation = sample.querySelector('.explanation');

                const userChoice = userAnswers[index];

                const correctChoice = correctAnswers[index];

               

                buttons.forEach(btn => {

                    btn.classList.remove('selected');

                    if (btn.dataset.choice === correctChoice) {

                        btn.classList.add('correct');

                    } else if (btn.dataset.choice === userChoice && userChoice !== correctChoice) {

                        btn.classList.add('incorrect');

                    }

                });

               

                if (userChoice === correctChoice) {

                    score++;

                }

               

                explanation.style.display = 'block';

            });

           

            // Show results

            const scoreElement = document.querySelector('.score');

            scoreElement.textContent = `You got ${score} out of 5 correct!`;

            results.style.display = 'block';

            submitBtn.style.display = 'none';

           

            // Add performance message

            let message = '';

            if (score === 5) message = 'Perfect! You have a great eye for detecting AI-generated content.';

            else if (score >= 3) message = 'Good work! You\'re developing strong critical evaluation skills.';

            else message = 'Don\'t worry - distinguishing AI from human writing takes practice!';

           

            results.querySelector('p').innerHTML = message + '<br><br>' + results.querySelector('p').innerHTML;

        }

 

        function resetQuiz() {

            userAnswers = [];

           

            samples.forEach(sample => {

                const buttons = sample.querySelectorAll('.choice-btn');

                buttons.forEach(btn => {

                    btn.classList.remove('selected', 'correct', 'incorrect');

                });

                sample.querySelector('.explanation').style.display = 'none';

            });

           

            results.style.display = 'none';

            submitBtn.style.display = 'block';

            submitBtn.disabled = true;

        }

    </script>
